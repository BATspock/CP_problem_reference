class Solution {
    public void moveZeroes(int[] nums) {
        Queue<Integer> q = new LinkedList<>();

        for (int i =0;i<nums.length;i++){
            if (nums[i]!=0){
                q.add(nums[i]);
                nums[i]=0;
            }
        }
        int index =0;
        while(! q.isEmpty()){
            int top = q.remove();
            nums[index] =top;
            index++;
        }
    }
}
